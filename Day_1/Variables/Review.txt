public long Fibonacci(int n)
{
    if (n < 0 || n >= 50)
    {
        throw new System.Exception("Not supported");
    }
 
    if (n == 0)
    {
        return 0;
    }
 
    if (n == 1)
    {
        return 1;
    }
 
    long a = 0, b = 1;
    for (int i = 2; i <= n; i++)
    {
        long temp = a + b;
        a = b;
        b = temp;
    }
 
    return b;
}
